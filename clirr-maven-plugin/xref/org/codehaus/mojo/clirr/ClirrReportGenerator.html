43       * Clirr's difference types.
44       */
45      private static final int FIELD_TYPE_CHANGED = 6004;
46      private static final int METHOD_ARGUMENT_TYPE_CHANGED = 7005;
47      private static final int METHOD_RETURN_TYPE_CHANGED = 7006;
48  
49      private static final class JustificationComparator implements Comparator<Difference>
50      {
51          public int compare( Difference o1, Difference o2 )
52          {
53              return o1.getJustification().compareTo( o2.getJustification() );
54          }
55      }
56  
57      private static final class ApiChangeComparator implements Comparator<ApiChange>
58      {
59          public int compare(ApiChange c1, ApiChange c2)
60          {
61              int cmp = c1.getAffectedClass().compareTo(c2.getAffectedClass());
62              if (cmp == 0)
63              {
64                  cmp = c1.getFrom().compareTo(c2.getFrom());
65                  if (cmp == 0)
66                  {
67                      return c1.getTo().compareTo(c2.getTo());
68                  }
69              }
70              return cmp;
71          }
72      }

74      private static class ApiChange
75      {
76          private Difference difference;
77          private List<ApiDifference> apiDifferences = new LinkedList<ApiDifference>();
78          private String from;
79          private String to;
80  
81          private String getAffectedClass()
82          {
83              return apiDifferences.get(0).getAffectedClass();
84          }
85  
86          private String getFrom()
87          {
88            return from;
89          }
90  
91          private String getTo()
92          {
93              return to;
94          }
95  
96          private void computeFields()
97          {
98            ApiDifference apiDiff = apiDifferences.get(0);
99            String methodSig = apiDiff.getAffectedMethod();
100 
101           // set default values that can be overwritten later
102           if (apiDiff.getAffectedMethod() != null)
103           {
104               from = apiDiff.getAffectedMethod();
105           }
106           else
107           {
108               from = apiDiff.getAffectedField();
109           }
110           to = difference.getTo();
111 
112 
113           switch (difference.getDifferenceType())
114           {
115           case FIELD_TYPE_CHANGED: {
116               String clirrReport = apiDiff.getReport(new MessageTranslator());
117               Pattern p = Pattern.compile("Changed type of field ([^ ]+) from ([^ ]+) to ([^ ]+)");
118               Matcher m = p.matcher(clirrReport);
119               if (m.find())
120               {
121                   from = m.group(2) + ' ' + m.group(1);
122                   to = m.group(3) + ' ' + m.group(1);
123               }
124               break;
125               }
126 
127           case METHOD_ARGUMENT_TYPE_CHANGED: {
128               to = Difference.getNewMethodSignature( methodSig, apiDifferences );
129               break;
130               }
131 
132           case METHOD_RETURN_TYPE_CHANGED: {
133               String clirrReport = apiDiff.getReport(new MessageTranslator());
134               Pattern p = Pattern.compile("Return type of method '[^']+' has been changed to (.+)");
135               Matcher m = p.matcher(clirrReport);
136               if (m.find())
137               {
138                   int openParIdx = methodSig.indexOf('(');
139                   int afterReturnTypeIdx = methodSig.lastIndexOf(' ', openParIdx);
140                   int beforeReturnTypeIdx = methodSig.lastIndexOf(' ', afterReturnTypeIdx - 1);
141                   to = new StringBuilder()
142                         .append(methodSig, 0, beforeReturnTypeIdx + 1)
143                         .append(m.group(1))
144                         .append(methodSig, afterReturnTypeIdx, methodSig.length())
145                         .toString();
146               }
147               break;
148               }
149 
150           }
151         }
152 
153     }

176     public void generateReport( ClirrDiffListener listener )
177     {
178         doHeading();
179 
180         if ( enableSeveritySummary )
181         {
182             doSeveritySummary( listener );
183         }
184 
185         doDetails( listener );
186         doApiChanges( listener );
187 
188         sink.body_();
189         sink.flush();
190         sink.close();
191     }

193     private void doHeading()
194     {
195         sink.head();
196         sink.title();
197 
198         String title = bundle.getString( "report.clirr.title" );
199         sink.text( title );
200         sink.title_();
201         sink.head_();
202 
203         sink.body();
204 
205         sink.section1();
206         sink.sectionTitle1();
207         sink.text( title );
208         sink.sectionTitle1_();
209 
210         sink.paragraph();
211         sink.text( bundle.getString( "report.clirr.clirrlink" ) + " " );
212         sink.link( "http://clirr.sourceforge.net/" );
213         sink.text( "Clirr" );
214         sink.link_();
215         sink.text( "." );
216         sink.paragraph_();
217 
218         sink.list();
219 
220         sink.listItem();
221         sink.text( bundle.getString( "report.clirr.version.current" ) + " " );
222         sink.text( getCurrentVersion() );
223         sink.listItem_();
224 
225         if ( getComparisonVersion() != null )
226         {
227             sink.listItem();
228             sink.text( bundle.getString( "report.clirr.version.comparison" ) + " " );
229             sink.text( getComparisonVersion() );
230             sink.listItem_();
231         }
232 
233         sink.list_();
234 
235         sink.section1_();
236     }

263     private void doSeveritySummary( ClirrDiffListener listener )
264     {
265         sink.section1();
266         sink.sectionTitle1();
267         sink.text( bundle.getString( "report.clirr.summary" ) );
268         sink.sectionTitle1_();
269 
270         sink.table();
271 
272         sink.tableRow();
273 
274         sink.tableHeaderCell();
275         sink.text( bundle.getString( "report.clirr.column.severity" ) );
276         sink.tableHeaderCell_();
277 
278         sink.tableHeaderCell();
279         sink.text( bundle.getString( "report.clirr.column.number" ) );
280         sink.tableHeaderCell_();
281 
282         sink.tableRow_();
283 
284         severityReportTableRow( listener, Severity.ERROR,
285             "report.clirr.level.error", "images/icon_error_sml.gif" );
286 
287         if ( minSeverity == null || minSeverity.compareTo( Severity.WARNING ) <= 0 )
288         {
289             severityReportTableRow( listener, Severity.WARNING,
290                 "report.clirr.level.warning", "images/icon_warning_sml.gif" );
291         }
292 
293         if ( minSeverity == null || minSeverity.compareTo( Severity.INFO ) <= 0 )
294         {
295             severityReportTableRow( listener, Severity.INFO,
296                 "report.clirr.level.info", "images/icon_info_sml.gif" );
297         }
298 
299         sink.table_();
300 
301         if ( minSeverity == null || minSeverity.compareTo( Severity.INFO ) > 0 )
302         {
303             sink.paragraph();
304             sink.italic();
305             sink.text( bundle.getString( "report.clirr.filtered" ) );
306             sink.italic_();
307             sink.paragraph_();
308         }
309 
310         sink.section1_();
311     }

328     private void doDetails( ClirrDiffListener listener )
329     {
330         sink.section1();
331         sink.sectionTitle1();
332         sink.text( bundle.getString( "report.clirr.api.incompatibilities" ) );
333         sink.sectionTitle1_();
334 
335         List<ApiDifference> differences = listener.getApiDifferences();
336 
337         if ( !differences.isEmpty() )
338         {
339             doIncompatibilitiesTable( differences );
340         }
341         else
342         {
343             sink.paragraph();
344             sink.text( bundle.getString( "report.clirr.noresults" ) );
345             sink.paragraph_();
346         }
347 
348         sink.section1_();
349     }

351     private void doIncompatibilitiesTable( List<ApiDifference> differences )
352     {
353         sink.table();
354         sink.tableRow();
355         sink.tableHeaderCell();
356         sink.text( bundle.getString( "report.clirr.column.severity" ) );
357         sink.tableHeaderCell_();
358         sink.tableHeaderCell();
359         sink.text( bundle.getString( "report.clirr.column.message" ) );
360         sink.tableHeaderCell_();
361         sink.tableHeaderCell();
362         sink.text( bundle.getString( "report.clirr.column.class" ) );
363         sink.tableHeaderCell_();
364         sink.tableHeaderCell();
365         sink.text( bundle.getString( "report.clirr.column.methodorfield" ) );
366         sink.tableHeaderCell_();
367         sink.tableRow_();
368 
369         MessageTranslator translator = new MessageTranslator();
370         translator.setLocale( locale );
371 
372         for ( ApiDifference difference : differences )
373         {
374             // TODO: differentiate source and binary? The only difference seems to be MSG_CONSTANT_REMOVED at this point
375             Severity maximumSeverity = difference.getMaximumSeverity();
376 
377             if ( minSeverity == null || minSeverity.compareTo( maximumSeverity ) <= 0 )
378             {
379                 sink.tableRow();
380 
381                 sink.tableCell();
382                 levelIcon( maximumSeverity );
383                 sink.tableCell_();
384 
385                 sink.tableCell();
386                 sink.text( difference.getReport( translator ) );
387                 sink.tableCell_();
388 
389                 sink.tableCell();
390                 if ( xrefLocation != null )
391                 {
392                     String pathToClass = difference.getAffectedClass().replace( '.', '/' );
393                     // MCLIRR-18 Special handling of links to inner classes:
394                     // We link to the page for the containing class
395                     final int innerClassIndex = pathToClass.lastIndexOf( '$' );
396                     if ( innerClassIndex != -1 )
397                     {
398                         pathToClass = pathToClass.substring( 0, innerClassIndex );
399                     }
400                     sink.link( xrefLocation + "/" + pathToClass + ".html" );
401                 }
402                 sink.text( difference.getAffectedClass() );
403                 if ( xrefLocation != null )
404                 {
405                     sink.link_();
406                 }
407                 sink.tableCell_();
408 
409                 sink.tableCell();
410                 sink.text( difference.getAffectedMethod() != null ? difference.getAffectedMethod()
411                     : difference.getAffectedField() );
412                 sink.tableCell_();
413 
414                 sink.tableRow_();
415             }
416         }
417 
418         sink.table_();
419     }

422     {

437     private void doApiChanges( ClirrDiffListener listener )
438     {
439         sink.section1();
440         sink.sectionTitle1();
441         sink.text( bundle.getString( "report.clirr.api.changes" ) );
442         sink.sectionTitle1_();
443 
444         Map<Difference, List<ApiChange>> apiChangeReport = getApiChangeReport( listener );
445         if ( apiChangeReport.isEmpty() )
446         {
447             sink.paragraph();
448             sink.text( bundle.getString( "report.clirr.noresults" ) );
449             sink.paragraph_();
450         }
451         else
452         {
453             doApiChangesTable( apiChangeReport );
454         }
455 
456         sink.section1_();
457     }

459     private Map<Difference, List<ApiChange>> getApiChangeReport( ClirrDiffListener listener )
460     {
461         final Map<String, List<ApiChange>> tmp = new HashMap<String, List<ApiChange>>();
462         for ( Entry<Difference, List<ApiDifference>> ignoredDiff
463             : listener.getIgnoredApiDifferences().entrySet() )
464         {
465             for ( ApiDifference apiDiff : ignoredDiff.getValue() )
466             {
467                 putApiChange( tmp, apiDiff, ignoredDiff.getKey() );
468             }
469         }
470         for ( ApiDifference apiDiff : listener.getApiDifferences() )
471         {
472             putApiChange( tmp, apiDiff, null );
473         }
474 
475 
476         final Map<Difference, List<ApiChange>> results =
477             new TreeMap<Difference, List<ApiChange>>( new JustificationComparator() );
478 
479         for ( List<ApiChange> changes : tmp.values() )
480         {
481             for ( ApiChange apiChange : changes )
482             {
483                 List<ApiChange> changesForDifference = results.get( apiChange.difference );
484                 if ( changesForDifference == null )
485                 {
486                     changesForDifference = new LinkedList<ApiChange>();
487                     results.put( apiChange.difference, changesForDifference );
488                 }
489                 changesForDifference.add( apiChange );
490             }
491         }
492         return results;
493     }

495     private void putApiChange( Map<String, List<ApiChange>> results,
496             ApiDifference ignoredDiff, Difference reason )
497     {
498         String apiChangeKey = getKey( ignoredDiff );
499         List<ApiChange> apiChanges = results.get( apiChangeKey );
500         if ( apiChanges == null )
501         {
502             apiChanges = new LinkedList<ApiChange>();
503             results.put(apiChangeKey, apiChanges);
504         }
505 
506         if ( reason != null && reason.getDifferenceType() == METHOD_ARGUMENT_TYPE_CHANGED )
507         {
508             ApiChange apiChange7005 = find7005ApiChange( apiChanges );
509             if ( apiChange7005 != null )
510             {
511                 apiChange7005.apiDifferences.add(ignoredDiff);
512                 return;
513             }
514         }
515         ApiChange change = new ApiChange();
516         change.difference = reason != null ? reason : createNullObject();
517         if (change.difference.getJustification() == null)
518         {
519           change.difference.setJustification( bundle.getString( "report.clirr.api.changes.unjustified" ) );
520         }
521         change.apiDifferences.add( ignoredDiff );
522         apiChanges.add( change );
523     }

525     /**
526      * Use a null object to avoid doing null checks everywhere.
527      */
528     private Difference createNullObject()
529     {
530         Difference difference = new Difference();
531         difference.setClassName("");
532         difference.setMethod("");
533         difference.setField("");
534         difference.setFrom("");
535         difference.setTo("");
536         difference.setJustification( bundle.getString( "report.clirr.api.changes.unjustified" ) );
537         return difference;
538     }

540     private String getKey( ApiDifference apiDiff )
541     {
542         if ( apiDiff.getAffectedMethod() != null )
543         {
544             return apiDiff.getAffectedClass() + " " + apiDiff.getAffectedMethod();
545         }
546         return apiDiff.getAffectedClass() + " " + apiDiff.getAffectedField();
547     }

549     private ApiChange find7005ApiChange( List<ApiChange> apiChanges )
550     {
551         for ( ApiChange apiChange : apiChanges )
552         {
553             if ( apiChange.difference.getDifferenceType() == METHOD_ARGUMENT_TYPE_CHANGED )
554             {
555                 return apiChange;
556             }
557         }
558         return null;
559     }

561     private void doApiChangesTable( Map<Difference, List<ApiChange>> apiChangeReport )
562     {
563         if ( comparisonVersion != null )
564         {
565             String[] args = new String[]{comparisonVersion, currentVersion};
566             String message = i18n.format(
567                 "clirr-report", locale, "report.clirr.api.changes.listing.comparisonversion", args );
568             sink.text( message );
569         }
570         else
571         {
572             sink.text( bundle.getString( "report.clirr.api.changes.listing" ) );
573         }
574 
575         sink.list();
576         for ( Entry<Difference, List<ApiChange>> apiChanges
577             : apiChangeReport.entrySet() )
578         {
579             sink.listItem();
580             sink.text( apiChanges.getKey().getJustification() );
581             sink.paragraph();
582 
583             sink.table();
584             sink.tableRow();
585             sink.tableHeaderCell();
586             sink.text( bundle.getString( "report.clirr.api.changes.class" ) );
587             sink.tableHeaderCell_();
588             sink.tableHeaderCell();
589             sink.text( bundle.getString( "report.clirr.api.changes.from" ) );
590             sink.tableHeaderCell_();
591             sink.tableHeaderCell();
592             sink.text( bundle.getString( "report.clirr.api.changes.to" ) );
593             sink.tableHeaderCell_();
594             sink.tableRow_();
595 
596             for (ApiChange apiChange : apiChanges.getValue())
597             {
598                 apiChange.computeFields();
599             }
600             Collections.sort(apiChanges.getValue(), new ApiChangeComparator());
601 
602             for (ApiChange apiChange : apiChanges.getValue())
603             {
604                 sink.tableRow();
605                 sink.tableCell();
606                 sink.text( apiChange.getAffectedClass() );
607                 sink.tableCell_();
608                 sink.tableCell();
609                 sink.text( apiChange.getFrom() );
610                 sink.tableCell_();
611                 sink.tableCell();
612                 sink.text( apiChange.getTo() );
613                 sink.tableCell_();
614                 sink.tableRow_();
615             }
616             sink.table_();
617             sink.paragraph_();
618             sink.listItem_();
619         }
620         sink.list_();
621     }
